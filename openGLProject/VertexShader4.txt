#version 330 core
//vertex shader has 15 min in variable slots, so we have to say which slot each variable is to use
layout (location = 0) in vec3 aPos; // in variable are passed into this shader
layout (location = 1) in vec3 aColor;
layout (location = 2) in vec2 aTexCoord;


out vec3 vertexColor;
out vec2 TexCoord;

void main(){
//in vertex shader, gl_Position must be set a value of type vec4
//this is to help with drawing/building primitive etc. so it need
//to know how to use it

gl_Position = vec4(aPos, 1.0);// or vec4(aPos.x, aPos.y, aPos.z, 1.0);

//vec3- has 3 values, you can represent them as xyz, rgb or rst
//vec4: has 4 values,xyzw and rgba

vertexColor = aColor;
TexCoord = aTexCoord;// pass it into the framgent shade
}
